function generarRecibosPDF() { const seleccionados = Array.from(document.querySelectorAll('.chk-recibo:checked')); if(seleccionados.length === 0) { alert('Seleccioná al menos a una persona'); return; } // Calcular exactamente la fecha del viernes de esta semana en pantalla (Sábado + 6 días) let h_hoy = new Date(); let sabadoInicial = getSabadoPrevio(h_hoy); sabadoInicial.setDate(sabadoInicial.getDate() + semOff * 7); let viernesObj = new Date(sabadoInicial); viernesObj.setDate(sabadoInicial.getDate() + 6); let diaV = viernesObj.getDate().toString().padStart(2, '0'); let mesV = (viernesObj.getMonth() + 1).toString().padStart(2, '0'); let anoV = viernesObj.getFullYear(); let fechaViernesFormateada = `${diaV}/${mesV}/${anoV}`; const contenedor = document.createElement('div'); contenedor.className = 'recibo-print-container'; // Inyección de estilos CSS puros para impresión A4 exacta y perfecta alineación let html = ` `; seleccionados.forEach((chk, index) => { const nombre = chk.value; const monto = parseFloat(chk.getAttribute('data-monto')); const montoFormat = monto.toLocaleString('es-AR', {minimumFractionDigits: 2, maximumFractionDigits: 2}); html += `
Recibo de Dinero
Fecha: ${fechaViernesFormateada} Recibí de: IIBR Ingeniería SRL
Nombre:  ${nombre}
La suma de:  $ ${montoFormat}
En concepto de:  Mano de obra de hormigón armado para la obra: Sanchez de Bustamante 1165
Firma Aclaración DNI / CUIT
`; // Aplica un salto de página forzado e impecable cada exactamente 4 elementos if ((index + 1) % 4 === 0 && (index + 1) !== seleccionados.length) { html += '
'; } }); contenedor.innerHTML = html; document.getElementById('recibos-overlay').style.display = 'none'; const periodo = document.getElementById('tit-sem').textContent.replace(/ /g, '_'); const opt = { margin: 10, filename: `Recibos_Semanales_${periodo}.pdf`, image: { type: 'jpeg', quality: 0.98 }, html2canvas: { scale: 2, logging: false, useCORS: true }, jsPDF: { unit: 'mm', format: 'a4', orientation: 'portrait' } }; html2pdf().set(opt).from(contenedor).save(); }
